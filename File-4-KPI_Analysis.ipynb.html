# Monthly KPIs
monthly_sales = df_sales.groupby("YearMonth")["Quantity"].sum()
monthly_returns = df_returns.groupby("YearMonth")["Quantity"].sum().abs()

monthly_return_rate = (monthly_returns / monthly_sales) * 100

# datetime index
monthly_return_rate.index = monthly_return_rate.index.to_timestamp()
monthly_sales.index = monthly_sales.index.to_timestamp()
monthly_returns.index = monthly_returns.index.to_timestamp()

plt.figure(figsize=(12,6))

3) Customer Return Ratio

In [39]:
# Sales quantity per customer
sales_cust = df[df["Quantity"] > 0] \
.groupby("CustomerID")["Quantity"].sum()

# Return quantity per customer
return_cust = df[df["Quantity"] < 0] \
.groupby("CustomerID")["Quantity"].sum().abs()

# Align and calculate ratio
customer_return_ratio = (return_cust / sales_cust).fillna(0)
top_customers = customer_return_ratio.sort_values(ascending=False).head(10).reset_index()

plt.figure(figsize=(10,6))
sns.barplot(
    data=top_customers,
    x="Quantity",
    y="CustomerID"
)

plt.title("Top 10 Customers by Return Ratio")
plt.xlabel("Return Ratio")
plt.ylabel("Customer")

plt.tight_layout()
plt.show()

Inference

  • The analysis highlights customers with the highest return ratios, indicating patterns of frequent returns and potential dissatisfaction
  • Identifying such customers helps in reducing return-related losses and improving customer management strategies
